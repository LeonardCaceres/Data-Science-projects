def get_mde(alpha, beta, sample_size, var_x):
    '''–†–∞—Å—á–∏—Ç—ã–≤–∞–µ—Ç MDE

    :param alpha: –∂–µ–ª–∞–µ–º–∞—è –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –æ—à–∏–±–∫–∏ –ø–µ—Ä–≤–æ–≥–æ —Ä–æ–¥–∞
    :param beta: –∂–µ–ª–∞–µ–º–∞—è –º–æ—â–Ω–æ—Å—Ç—å
    :param sample_size: —Ä–∞–∑–º–µ—Ä –≤—ã–±–æ—Ä–æ–∫
    :param var_x, var_y: –¥–∏—Å–ø–µ—Ä—Å–∏–∏ –≤—ã–±–æ—Ä–æ–∫
    :returns: —Ç–µ–æ—Ä–µ—Ç–∏—á–µ—Å–∫–∏–π MDE
    '''
    var_y=var_x
    q_sum = sps.norm.ppf(1 - alpha) + sps.norm.ppf(beta)
    return q_sum / np.sqrt(sample_size) * np.sqrt(var_x + var_y)

def absolute_ttest(x, y):
    """
    –ê–±—Å–æ–ª—é—Ç–Ω—ã–π t-test.

    –ê—Ä–≥—É–º–µ–Ω—Ç—ã:
    x, y -- –≤—ã–±–æ—Ä–∫–∏ –æ–¥–∏–Ω–∞–∫–æ–≤–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞

    """
    –û—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã–π t-test.

    –ê—Ä–≥—É–º–µ–Ω—Ç—ã:
    x, y -- –≤—ã–±–æ—Ä–∫–∏ –æ–¥–∏–Ω–∞–∫–æ–≤–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞
    alpha -- —É—Ä–æ–≤–µ–Ω—å –∑–Ω–∞—á–∏–º–æ—Å—Ç–∏

    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
    stat -- —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –∫—Ä–∏—Ç–µ—Ä–∏—è
    pvalue
    left_bound, right_bound -- –≥—Ä–∞–Ω–∏—Ü—ã –¥–æ–≤. –∏–Ω—Ç–µ—Ä–≤–∞–ª–∞
    """

    n = len(x)
    x_mean = x.mean()
    y_mean = y.mean()

    stat = x_mean / y_mean - 1
    var = x.var() / (y_mean**2) + y.var() * (x_mean**2) / (y_mean**4)
    std = np.sqrt(var)

    z_stat = np.sqrt(n) * stat / std
    pvalue = 2 * sps.norm.sf(np.abs(z_stat))

    q = sps.norm.ppf(1 - alpha / 2)
    left_bound = stat - q * std / np.sqrt(n)
    right_bound = stat + q * std / np.sqrt(n)

    return stat, pvalue, left_bound, right_bound
def estimate_reject_prob(n_rejects, n_iter):
    """
    –û—Ü–µ–Ω–∫–∞ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –æ—Ç–≤–µ—Ä–∂–µ–Ω–∏—è –∫—Ä–∏—Ç–µ—Ä–∏—è –∏ –µ–µ –¥–æ–≤. –∏–Ω—Ç–µ—Ä–≤–∞–ª–∞.
    –ò—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è –æ—Ü–µ–Ω–∫–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –æ—à–∏–±–∫–∏ –ø–µ—Ä–≤–æ–≥–æ —Ä–æ–¥–∞ –∏ –º–æ—â–Ω–æ—Å—Ç–∏

    –ê—Ä–≥—É–º–µ–Ω—Ç—ã:
    n_rejects -- –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –æ—Ç–≤–µ—Ä–∂–µ–Ω–∏–π H_0 –≤ —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–µ
    n_iter -- –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–æ–≤

    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
    prob_reject -- –æ—Ü–µ–Ω–∫–∞ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –æ—Ç–≤–µ—Ä–∂–µ–Ω–∏—è –∫—Ä–∏—Ç–µ—Ä–∏—è
    left_bound, right_bound -- –≥—Ä–∞–Ω–∏—Ü—ã —Å–æ–æ—Ç–≤. –¥–æ–≤. –∏–Ω—Ç–µ—Ä–≤–∞–ª–∞
    """

    prob_reject = n_rejects / n_iter
    left_bound, right_bound = proportion_confint(n_rejects, n_iter, method="wilson")

    return prob_reject, left_bound, right_bound

def add_effect(x, effect, target_name="pilot", relative_effect=True):
    """
    –§—É–Ω–∫—Ü–∏—è –∏—Å–∫—É—Å—Å—Ç–≤–µ–Ω–Ω–æ–≥–æ –¥–æ–±–∞–≤–ª–µ–Ω–∏—è —ç—Ñ—Ñ–µ–∫—Ç–∞ –≤ —Ç–µ—Å—Ç–æ–≤—É—é –≤—ã–±–æ—Ä–∫—É

    –ê—Ä–≥—É–º–µ–Ω—Ç—ã:
    x -- –≤—ã–±–æ—Ä–∫–∞
    effect -- –¥–æ–±–∞–≤–ª—è–µ–º—ã–π —ç—Ñ—Ñ–µ–∫—Ç

    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
    x -- –≤—ã–±–æ—Ä–∫–∞ —Å –¥–æ–±–∞–≤–ª–µ–Ω–Ω—ã–º —ç—Ñ—Ñ–µ–∫—Ç–æ–º
    """

    x_copy = x.copy()
    if relative_effect:
        x_copy[target_name] *= 1 + effect
    else:
        x_copy[target_name] += effect

    return x_copy


def run_1_iteration(
    test,
    data,
    sample_size,
    generate_samples,
    target_name,
    add_effect,
    effects_list=[0],
    relative_effect=True,
):
    """
    –ü—Ä–æ–≤–µ–¥–µ–Ω–∏–µ —Å–µ—Ä–∏–∏ AA-—Ç–µ—Å—Ç–æ–≤ –Ω–∞ –∏—Å–∫—É—Å—Å—Ç–≤–µ–Ω–Ω—ã—Ö –≤—ã–±–æ—Ä–∫–∞—Ö.

    –ê—Ä–≥—É–º–µ–Ω—Ç—ã:
    test -- —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏–π –∫—Ä–∏—Ç–µ—Ä–∏–π
    generate_samples -- —Ñ—É–Ω–∫—Ü–∏—è –¥–ª—è —Å–µ–º–ø–ª–∏—Ä–æ–≤–∞–Ω–∏—è –≤—ã–±–æ—Ä–æ–∫
    add_effect -- —Ñ—É–Ω–∫—Ü–∏—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è —ç—Ñ—Ñ–µ–∫—Ç–∞
    effects_list --  –º–∞—Å—Å–∏–≤ —Ä–∞–∑–º–µ—Ä–æ–≤ –¥–æ–±–∞–≤–ª—è–µ–º–æ–≥–æ —ç—Ñ—Ñ–µ–∫—Ç–∞ (–¥–ª—è –æ—Ü–µ–Ω–∫–∏ –º–æ—â–Ω–æ—Å—Ç–∏)
    relative_effect -- —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —ç—Ñ—Ñ–µ–∫—Ç –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã–º –∏–ª–∏ –∞–±—Å–æ–ª—é—Ç–Ω—ã–º

    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç: pvalue
    """
    # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –≤—ã–±–æ—Ä–∫–∏
    x_data, y_data = generate_samples(data, sample_size)

    pvals = []
    for effect in effects_list:
        # –î–æ–±–∞–≤–ª—è–µ–º —ç—Ñ—Ñ–µ–∫—Ç, –µ—Å–ª–∏ —Ö–æ—Ç–∏–º –æ—Ü–µ–Ω–∏—Ç—å –º–æ—â–Ω–æ—Å—Ç—å
        x_data_cp = add_effect(
            x_data, effect, target_name=target_name, relative_effect=relative_effect
        )
        # –ü—Ä–∏–º–µ–Ω—è–µ–º –∫—Ä–∏—Ç–µ—Ä–∏–π
        pvals.append(test(x_data_cp, y_data)[1])
    return pvals


def run_experiments(
    test,
    generate_samples,
    data=None,
    add_effect=add_effect,
    n_iter=10000,
    sample_size=1000,
    effect=0,
    relative_effect=True,
    n_jobs=4,
    alpha=0.05,
    target_name="pilot",
    draw=False,
    title=None,
    show_pvals=True,
):
    """
    –ü—Ä–æ–≤–µ–¥–µ–Ω–∏–µ —Å–µ—Ä–∏–∏ AA-—Ç–µ—Å—Ç–æ–≤ –Ω–∞ –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏—Ö –¥–∞–Ω–Ω—ã—Ö.

    –ê—Ä–≥—É–º–µ–Ω—Ç—ã:
    test -- —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏–π –∫—Ä–∏—Ç–µ—Ä–∏–π
    generate_samples -- —Ñ—É–Ω–∫—Ü–∏—è –¥–ª—è —Å–µ–º–ø–ª–∏—Ä–æ–≤–∞–Ω–∏—è –≤—ã–±–æ—Ä–æ–∫
    data -- –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏–µ –¥–∞–Ω–Ω—ã–µ
    n_iter -- –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π
    sample_size -- —Ä–∞–∑–º–µ—Ä –≤—ã–±–æ—Ä–æ–∫
    add_effect -- —Ñ—É–Ω–∫—Ü–∏—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è —ç—Ñ—Ñ–µ–∫—Ç–∞
    effect -- —Ä–∞–∑–º–µ—Ä –¥–æ–±–∞–≤–ª—è–µ–º–æ–≥–æ —ç—Ñ—Ñ–µ–∫—Ç–∞ (–¥–ª—è –æ—Ü–µ–Ω–∫–∏ –º–æ—â–Ω–æ—Å—Ç–∏)
    relative_effect -- —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —ç—Ñ—Ñ–µ–∫—Ç –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã–º –∏–ª–∏ –∞–±—Å–æ–ª—é—Ç–Ω—ã–º
    alpha -- —Ç–µ–æ—Ä –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –æ—à–∏–±–∫–∏ –ø–µ—Ä–≤–æ–≥–æ —Ä–æ–¥–∞
    target_name -- –∏–º—è –∫–æ–ª–æ–Ω–∫–∏ —Å —Ç–∞—Ä–≥–µ—Ç-–º–µ—Ç—Ä–∏–∫–æ–π
    """

    # –ü—Ä–æ–≤–µ–¥–µ–Ω–∏–µ —Å–µ—Ä–∏–∏ —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–æ–≤
    pvals = Parallel(n_jobs=n_jobs)(
        delayed(run_1_iteration)(
            test, data, sample_size, generate_samples, target_name, add_effect, [effect], relative_effect
        )
        for _ in tqdm(range(n_iter), leave=False)
    )
    pvals = np.array(pvals).flatten()
    # –ü–æ–¥—Å—á–µ—Ç —á–∏—Å–ª–∞ –æ—Ç–≤–µ—Ä–∂–µ–Ω–∏–π
    n_rejects = (pvals < alpha).sum()

    # –û—Ü–µ–Ω–∫–∞ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –æ—Ç–≤–µ—Ä–∂–µ–Ω–∏—è
    prob_reject, left_bound, right_bound = estimate_reject_prob(n_rejects, n_iter)

    # –í–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è
    if draw:
        figsize = (14, 3) if show_pvals else (5, 2)
        visualization(
            prob_reject,
            left_bound,
            right_bound,
            show_pvals,
            pvals,
            alpha,
            figsize,
            title,
        )

    return prob_reject, (left_bound, right_bound)

    alpha=0.05,

    """
    –ü—Ä–æ–≤–µ–¥–µ–Ω–∏–µ —Å–µ—Ä–∏–∏ AA-—Ç–µ—Å—Ç–æ–≤ –Ω–∞ –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏—Ö –¥–∞–Ω–Ω—ã—Ö —Å –¥–æ–±–∞–≤–µ–Ω–∏–µ–º —Ä–∞–∑–Ω—ã—Ö —ç—Ñ—Ñ–µ–∫—Ç–æ–≤.

    –ê—Ä–≥—É–º–µ–Ω—Ç—ã:
    test -- —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏–π –∫—Ä–∏—Ç–µ—Ä–∏–π
    generate_samples -- —Ñ—É–Ω–∫—Ü–∏—è –¥–ª—è —Å–µ–º–ø–ª–∏—Ä–æ–≤–∞–Ω–∏—è –≤—ã–±–æ—Ä–æ–∫
    data -- –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏–µ –¥–∞–Ω–Ω—ã–µ
    n_iter -- –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π
    sample_size -- —Ä–∞–∑–º–µ—Ä –≤—ã–±–æ—Ä–æ–∫
    add_effect -- —Ñ—É–Ω–∫—Ü–∏—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è —ç—Ñ—Ñ–µ–∫—Ç–∞
    effect -- —Ä–∞–∑–º–µ—Ä –¥–æ–±–∞–≤–ª—è–µ–º–æ–≥–æ —ç—Ñ—Ñ–µ–∫—Ç–∞ (–¥–ª—è –æ—Ü–µ–Ω–∫–∏ –º–æ—â–Ω–æ—Å—Ç–∏)
    relative_effect -- —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —ç—Ñ—Ñ–µ–∫—Ç –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω—ã–º –∏–ª–∏ –∞–±—Å–æ–ª—é—Ç–Ω—ã–º
    alpha -- —Ç–µ–æ—Ä –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –æ—à–∏–±–∫–∏ –ø–µ—Ä–≤–æ–≥–æ —Ä–æ–¥–∞
    target_name -- –∏–º—è –∫–æ–ª–æ–Ω–∫–∏ —Å —Ç–∞—Ä–≥–µ—Ç-–º–µ—Ç—Ä–∏–∫–æ–π
    """
    # –ü—Ä–æ–≤–µ–¥–µ–Ω–∏–µ —Å–µ—Ä–∏–∏ —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–æ–≤
    pvals = Parallel(n_jobs=n_jobs)(

    n_rejects = (np.array(pvals) < alpha).sum(axis=0)

    powers, left_bounds, right_bounds = estimate_reject_prob(n_rejects, n_iter)
